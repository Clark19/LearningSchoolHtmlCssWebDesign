<!-- HTML 부분 -->

<input id="close-btn" type="checkbox" /> <div class="menu-container"> <label class="on-off" for="sidebar">X</label> <!-- 원하는 것들 넣을 자리 --> </div>

<!-- CSS 부분 -->

// 체크박스 상자 자체는 보일 필요가 없음. #close-btn { display: none; } ​ .menu-container { // 체크되지 않았을 때 CSS } ​ //체크박스(레이블 텍스트)가 체크 되었을 때, 형재 엘리먼트 .menu-container를 지정해 스타일 설정 #close-btn:checked ~ .menu-container { // 체크되었을 때 스타일 }
